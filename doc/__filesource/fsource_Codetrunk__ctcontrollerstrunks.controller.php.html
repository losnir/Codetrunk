  71.             }
  72.             $errorParams = array($ctName, $ctParent['Code'], array('Submit a correction to a trunk by <a href="'.$ctParent['Url'].'"><u>'.$ctParent['Name'].'</u></a>', $ctPKey, $ctParent['Syntax']));

  405.          $recentTrunks[$trunkId]['ageFmt'] = $trunkAge;
