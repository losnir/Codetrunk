  52.       $cWidth  = 95;

  54.       $cGSpace = 4;

  57.       $cBColor = imagecolorallocate($cImage, 232, 244, 243);
  58.       $cGColor = imagecolorallocate($cImage, 194, 219, 233);
  59.       $cTColor = imagecolorallocate($cImage, 31, 68, 102);
  60.       imageline($cImage, 0, 0, $cWidth, 0, $cGColor);
  61.       for ($i=0; $i<= $cWidth; $i += $cGSpace) imageline($cImage, $i, 0, $i, $cHeight, $cGColor);
  62.       for ($i=0; $i<= $cHeight;  $i += $cGSpace) imageline($cImage, 0, $i, $cWidth, $i, $cGColor);
  63.       imagettftext($cImage, 22, 5, 5, 24, $cTColor, _CP."/ct/fonts/monofont.ttf", $cString);
  64.       $_SESSION['CodetrunkCaptcha'] = $cString;
