  91.    function followRoute($Url, $Cap = array(), $Params = array()) {
  92.       foreach(array_reverse($this->Routes) AS $Route)
  93.          if(!preg_match("/^{$Route[0]}$/", $Url, $Matches)) continue; else { $Cap = $Route; break; };
