  44.     * The constructor takes one parameter which is the storage dir

  49.    function __construct($storageDir) {
  50.       $this->storageDir = $storageDir;
  51.       if(!is_writable($this->storageDir)) { trigger_error("The directory '{$this->storageDir}' is not writable !!!",  E_USER_ERROR); exit; }

  54.    /**
  55.     * Fetches the trunk full path for the specified trunk key
  56.     * 
  57.     * File::getTrunkPath()

  60.     */
  61.    function getTrunkPath($trunkKey) {
  62.       $trunkPath = implode("/", array($this->storageDir, substr($trunkKey, 0, 1), substr($trunkKey, 1, 2), substr($trunkKey, 3, 2), substr($trunkKey, 5, 2), $trunkKey));
  63.       return (is_dir(dirname($trunkPath)) ? $trunkPath : (mkdir(dirname($trunkPath), 0777, true) ? $trunkPath : false));
  64.    }

  66.    /**
  67.     * Fetches the MRU (Most Recent Updated) for the specified domain
  68.     * 
  69.     * File::getMru()
  70.     * @param string $Domain
  71.     * @return string
  72.     */
  73.    function getMru($Domain) {
  74.       if(!$l = strlen($Domain)) {
  75.          $mruPath = implode("/", array($this->storageDir, "mru", "default.mru"));
  76.          return (is_dir(dirname($mruPath)) ? $mruPath : (mkdir(dirname($mruPath), 0777, true) ? $mruPath : false));
  77.       } else {
  78.          $mruPath = implode("/", array($this->storageDir, "mru"));
  79.          for($p = 0; $p < min(3, $l); $p++) $mruPath = implode("/", array($mruPath, substr($Domain, $p, 1)));
  80.          $mruPath = implode("/", array($mruPath, "{$Domain}.mru"));
  81.          return (is_dir(dirname($mruPath)) ? $mruPath : (mkdir(dirname($mruPath), 0777, true) ? $mruPath : false));
  82.       }
  83.    }

  85.    /**
  86.     * Updates the MRU File
  87.     * If $updateMru is an array then it will be shifted (prepend) to the MRU,
  88.     * Else it should be string containing a trunk key to be deleted from the MRU.
  89.     * 
  90.     * File::updateMru()
  91.     * @param string $Domain
  92.     * @param mixed $updateMru See the description above
  93.     */
  94.    function updateMru($Domain, $updateMru) {
  95.       $mruFile = $this->getMru($Domain);
  96.       $mruOpen = fopen($mruFile, "a+");
  97.       if($mruOpen) {
  98.          if(flock($mruOpen, LOCK_EX)) {
  99.             $mruData = false;
  100.             while(!feof($mruOpen)) { $mruData .= fgets($mruOpen, 4096); }
  101.             if(strlen($mruData)) { $newMru = unserialize($mruData); if(count($newMru) > 15) array_pop($newMru); } else $newMru = array();
  102.             if(is_array($updateMru)) array_unshift($newMru, $updateMru);
  103.             else if(Codetrunk::getInstance()->getController("Trunks")->getTrunkKey($updateMru))
  104.                foreach($newMru as $mruId => $mruValue) if($mruValue['Key'] == $updateMru) { unset($newMru[$mruId]); break; }
  105.             ftruncate($mruOpen, 0);
  106.             fwrite($mruOpen, serialize($newMru));
  107.             flock($mruOpen, LOCK_UN);
  108.             fclose($mruOpen);   
  109.          }
  110.       } else trigger_error("Could not open Most Recent Updated file '{$mruFile}' !!!", E_WARNING);
  111.    }

  113.    /**
  114.     * Fetches a time string (formatted) using strftime

  125.    /**
  126.     * Adds a trunk
  127.     * 
  128.     * File::addTrunk()
  129.     * @param string $Name Trunk User Name
  130.     * @param string $Domain Domain
  131.     * @param string $Syntax Syntax (language key)
  132.     * @param string $Code Code
  133.     * @param string $pKey Parent Trunk Keu. Use "" for a fresh trunk.
  134.     * @param string $expiryFlag Expiry flag. Could be one of d / m / f
  135.     * @param string $Token Unique token for ownership
  136.     * @return string
  137.     */
  138.    function addTrunk($Name, $Domain, $Syntax, $Code, $pKey, $expiryFlag, $Token) {
  139.       $Time = time();
  140.       $timeString = $this->getTimeString($Time);
  141.       if(!in_array($expiryFlag, array("d", "m", "f"))) $expiryFlag = Codetrunk::getInstance()->Config['Expiry'];
  142.       switch($expiryFlag) {
  143.          case "d": $expireTime = $Time + 86400; break;
  144.          case "m": $expireTime = $Time + 2592000; break;
  145.          case "f": $expireTime = 0; break;
  146.       }
  147.       do {
  148.          $trunkKey        = $expiryFlag.sprintf("%07x", mt_rand(0, 0xFFFFFFF));
  149.          $fileOpen        = fopen($this->getTrunkPath($trunkKey), "x");
  150.       } while($fileOpen === false);
  151.       if($fileOpen) {
  152.          $newData = func_get_args(); array_push($newData, array(), $Time, $timeString, $expireTime, $trunkKey);
  153.          fwrite($fileOpen, serialize(array_combine(array("Name", "Domain", "Syntax", "Code", "pKey", "expiryFlag", "Token", "followUps", "Time", "timeString", "expireTime", "Key"), $newData)));
  154.          fclose($fileOpen);
  155.          if(strlen($pKey)) {
  156.             $followUp['Key']        = $trunkKey;
  157.             $followUp['Name']       = $Name;
  158.             $followUp['postFormat'] = $timeString;
  159.             $parentFile = $this->getTrunkPath($pKey);
  160.             $parentOpen = fopen($parentFile, "a+");
  161.             if($parentOpen) {
  162.                if(flock($parentOpen, LOCK_EX)) {
  163.                   $parentData = false;
  164.                   while(!feof($parentOpen)) { $parentData .= fread($parentOpen, 4096); }
  165.                   if(strlen($parentData)) $newParent = unserialize($parentData); else trigger_error("Invalid parent trunk at '{$parentFile}' !!!", E_ERROR);
  166.                   $newParent['followUps'][] = $followUp;
  167.                   ftruncate($parentOpen, 0);
  168.                   fwrite($parentOpen, serialize($newParent));
  169.                   flock($parentOpen, LOCK_UN);
  170.                   fclose($parentOpen);
  171.                }
  172.             } else trigger_error("Could not open parent trunk at '{$parentFile}' !!!", E_WARNING);
  173.          }
  174.       }
  175.       $updateMru['Key']        = $trunkKey;
  176.       $updateMru['Name']       = $Name;
  177.       $updateMru['Time']       = $Time;
  178.       $updateMru['postFormat'] = $timeString;
  179.       $this->updateMru($Domain, $updateMru);
  180.       return $trunkKey;
  181.     }

  191.    function getTrunk($trunkKey, $Domain) {
  192.       $trunkFile = $this->getTrunkPath($trunkKey);
  193.       if(file_exists($trunkFile)) {
  194.          $trunkOpen = fopen($trunkFile, "r");
  195.          if($trunkOpen) {
  196.             $trunkData = false;
  197.             while(!feof($trunkOpen)) { $trunkData .= fread($trunkOpen, 4096); }
  198.             if(strlen($trunkData)) $newTrunk = unserialize($trunkData); else trigger_error("Invalid trunk at '{$trunkFile}' !!!", E_ERROR);
  199.             fclose($trunkOpen);   
  200.          } else trigger_error("Could not open trunk at '{$trunkFile}' !!!", E_ERROR);
  201.          $newTrunk['Url'] = Codetrunk::getInstance()->getController("Trunks")->getTrunkUrl($newTrunk['Key']);
  202.          if($newTrunk['Domain'] != $Domain) return false;
  203.          elseif((time() > $newTrunk['expireTime']) && $newTrunk['expireTime'] != 0) return false;
  204.          else return $newTrunk;
  205.       } else return false;
  206.    }

  209.     * Fetches recent trunks from MRU

  217.       $mruFile = $this->getMru($Domain);
  218.        if(file_exists($mruFile)) {
  219.          $mruOpen = fopen($mruFile, "a+");
  220.          if($mruOpen) {

  222.             while(!feof($mruOpen)) { $mruData .= fgets($mruOpen, 4096); }
  223.             if(strlen($mruData)) { $newMru = unserialize($mruData); while(count($newMru) > $Count) array_pop($newMru); } else trigger_error("Invalid MRU at '{$mruFile}' !!!", E_WARNING);
  224.             fclose($mruOpen);   
  225.          } else trigget_error("Could not open Most Recent Updated file '{$mruFile}' !!!", E_WARNING);
  226.           foreach($newMru as $mruId => $mruValue) $newMru[$mruId]['Age'] = time() - $mruValue['Time'];;
  227.        } else $newMru = array();   
  228.        return $newMru;
  229.    }
